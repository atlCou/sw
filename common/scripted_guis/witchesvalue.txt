scripted_gui = {
	
	manpower = {
		window_name = "witches_container"
		context_type = player_context
		visible = { always = yes }
	}
}

scripted_gui = {
	
	assigning_witches = {
		window_name = "assigning_witches_buttons"
		context_type = selected_state_context
		visible = { 
			OR = {
				if = { ROOT = { has_country_flag = unthinkable_witches_option }
					witches_manpower > 1 }
				if = { 
					ROOT = { 
						NOT = { 
							has_country_flag = unthinkable_witches_option 
						}
					}
					is_owned_by = NRO 
					witches_manpower > 1
				}	
			}
		}
		effects = {
			add_click = {
				NRO = { 
					any_owned_state = {
						add_state_modifier = {
							modifier = {
								local_org_regain = -100
								attrition = 0.05
							}
						}
					}	
				}	
				ROOT = { 
					add_to_variable = {
						witches_manpower = -1
					}
					PREV = {
						add_to_variable = {
							var_assigning_witches@PREV = 1000
						}
					}
				}
			}
		}
		effects = {
			remove_click = {
				ROOT = { 
					add_to_variable = {
						witches_manpower = 1
					}
					PREV = {
						add_to_variable = {
							var_assigning_witches@PREV = -1000
						}

					}
				}
			}
		}
	}
}
